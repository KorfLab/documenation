164// hashing materials
165static double HASH_MULTIPLIER[7] = {
166 3.1415926536, // PI
167 2.7182818285, // e
168 1.6180339887, // golden mean
169 1.7320508076, // square root of 3
170 2.2360679775, // square root of 5
171 2.6457513111, // square root of 7
172 3.3166247904 // square root of 11
173};
174static double MAX_HASH_DEPTH = 2.0;
175static int HashLevelToSlots(int level) {return pow(4, level);}
176static int HashFunc(const gkn_map hash, const char *key) {
177 double sum = 0;
178 for (int i = 0; i < strlen(key); i++)
179 sum += key[i] * HASH_MULTIPLIER[i % 7];
180 return (int) (hash->slots * (sum - floor(sum)));
181}
182
183static void ExpandHash(gkn_map hash) {
184 int oldslots = hash->slots;
185 gkn_vec *oldkey = hash->key;
186 gkn_vec *oldval = hash->val;
187 gkn_vec kvec;
188 gkn_vec vvec;
189 gkn_tvec keys;
190
191 // create the new hash
192 hash->level = hash->level +1;
193 hash->slots = HashLevelToSlots(hash->level);
194 hash->key = gkn_malloc(hash->slots * sizeof(struct gkn_VEC));
195 hash->val = gkn_malloc(hash->slots * sizeof(struct gkn_VEC));
196 for (int i = 0; i < hash->slots; i++) {
197 hash->key[i] = gkn_vec_new();
198 hash->val[i] = gkn_vec_new();
199 }
200
201 // brand new hash?
202 if (hash->keys->size == 0) return;
203
204 keys = hash->keys;
205 hash->keys = gkn_tvec_new();
206
207 // transfer old stuff to new hash
208 for (int i = 0; i < oldslots; i++) {
209 kvec = oldkey[i];
210 vvec = oldval[i];
211 for (int j = 0; j < kvec->size; j++) {
212 char *key = kvec->elem[j];
213 char *val = vvec->elem[j];
214 gkn_map_set(hash, key, val);
215 }
216 }
217
218 // free old stuff
219 for (int i = 0; i < oldslots; i++) {
220 kvec = oldkey[i];
221 vvec = oldval[i];
222 gkn_vec_free(kvec);
223 gkn_vec_free(vvec);
224 }
225 free(oldkey);
226 free(oldval);
227 gkn_tvec_free(keys);
228}

255void * gkn_map_get(const gkn_map hash, const char *key) {
256 int index = HashFunc(hash, key);
257
258 // resolve collisions
259 for (int i = 0; i < hash->key[index]->size; i++) {
260 char *string = hash->key[index]->elem[i];
261 if (strcmp(key, string) == 0) {
262 return hash->val[index]->elem[i];
263 }
264 }
265 return NULL; // return is NULL if not found
266}

268void gkn_map_set(gkn_map hash, const char *key, void *val) {
269 int new_key = 1;
270 int index = HashFunc(hash, key);
271
272 // reassign unless new key
273 for (int i = 0; i < hash->key[index]->size; i++) {
274 char *string = hash->key[index]->elem[i];
275 if (strcmp(key, string) == 0) {
276 hash->val[index]->elem[i] = val;
277 new_key = 0;
278 return;
279 }
280 }
281
282 if (new_key) {
283 gkn_tvec_push(hash->keys, key);
284 gkn_vec_push(hash->key[index], hash->keys->last);
285 gkn_vec_push(hash->vals, val);
286 gkn_vec_push(hash->val[index], hash->vals->last);
287 }
288
289 // check if we have to expand the hash
290 if ((double)hash->keys->size / (double)hash->slots >= MAX_HASH_DEPTH) {
291 ExpandHash(hash);
292 }
293}

307void gkn_map_stat(const gkn_map hash) {
308 int max = 0;
309 int min = INT_MAX;
310 int total = 0;
311 for (int i = 0; i < hash->slots; i++) {
312 int count = hash->val[i]->size;
313 total += count;
314 if (count > max) max = count;
315 if (count < min) min = count;

318 hash->level, hash->slots, hash->keys->size, min, max,

401void gkn_xtree_set (gkn_xtree xt, const char *string, void *value) {
402 int len = strlen(string);
403 if (len < 1) gkn_exit("gkn_xtree_set with empty string");
404 if (len >= MAX_WORD_LENGTH)
405 gkn_exit("gkn_xtree word length exceeded (%d)\n", MAX_WORD_LENGTH);
406
407 gkn_xnode parent = xt->head;
408 for (int i = 0; i < len; i++) {
409 char c = string[i];
410 gkn_xnode child = gkn_xnode_search(parent, c);
411 if (child == NULL) {
412 child = gkn_xnode_new(c);
413 gkn_vec_push(parent->children, child);
414 gkn_vec_push(xt->alloc, child);
415 }
416 parent = child;
417 }
418
419 parent->data = value;
420}

498void gkn_register_option(const char *name, int flag) {
499 if (COMMAND_LINE == NULL) {

503 }
504

506 case 0: gkn_map_set(CL_REGISTER, name, (void *)1); break;
507 case 1: gkn_map_set(CL_REGISTER, name, (void *)2); break;
508 default: gkn_exit("gkn_register_option: flag 0 or 1");

512void gkn_parse_options(int *argc, char **argv) {
513 for (int i = 0; i < *argc; i++) {
514 char *token = argv[i];
515 if (token[0] == '-' && strlen(token) > 1) {
516 switch ((size_t)gkn_map_get(CL_REGISTER, token)) {
517 case 0:
518 gkn_exit("unknown option (%s)", token);
519 break;
520 case 1:
521 gkn_map_set(CL_OPTIONS, token, token);
522 break;
523 case 2:
524 gkn_map_set(CL_OPTIONS, token, argv[i+1]);
525 i++;
526 break;
527 default:
528 gkn_exit("not possible");
529 }
530 } else {
531 gkn_tvec_push(COMMAND_LINE, argv[i]);
532 }
533 }
534
535 *argc = COMMAND_LINE->size;
536 for (int i = 0; i < COMMAND_LINE->size; i++) {
537 argv[i] = COMMAND_LINE->elem[i];
538 }
539}

556gkn_pipe gkn_pipe_open(const char *name, const char *mode) {
557 char command[1024];
558 int length = strlen(name);
559 gkn_pipe pipe = gkn_malloc(sizeof(struct gkn_PIPE));
560
561 if (strcmp(mode, "r") == 0) pipe->mode = 0;
562 else if (strcmp(mode, "w") == 0) pipe->mode = 1;
563 else if (strcmp(mode, "r+") == 0) pipe->mode = 2;
564 else gkn_exit("r, w, or r+ only in gkn_pipe");
565
566 pipe->name = gkn_malloc(length + 1);
567 strcpy(pipe->name, name);
568
569 pipe->gzip = 0;
570
571 if (name[length -3] == '.' &&
572 name[length -2] == 'g' &&
573 name[length -1] == 'z') pipe->gzip = 1; // .gz
574 if (name[length -2] == '.' &&
575 name[length -1] == 'z') pipe->gzip = 1; // .z
576 if (name[length -2] == '.' &&
577 name[length -1] == 'Z') pipe->gzip = 1; // .Z
578
579 if (pipe->gzip) {
580 if (pipe->mode != 0) gkn_exit("compressed pipes are read only");
581 sprintf(command, "gunzip -c %s", name);
582 pipe->stream = popen(command, "r");
583 } else {
584 if (strcmp(name, "-") == 0) pipe->stream = stdin;
585 else pipe->stream = fopen(name, mode);
586 }
587
588 if (pipe->stream == NULL) {
589 gkn_exit("failed to open %s\n", name);
590 }
591
592 return pipe;
593}

595char * gkn_readline(gkn_pipe io) {
596 char line[4096];
597 int read = 0;
598 while (fgets(line, sizeof(line), io->stream) != NULL) {
599 if (line[0] == '#') continue; // skipping comments
600 if (strlen(line) == 0) continue;
601 read = 1;
602 break;
603 }
604 if (read == 0) return NULL;
605
606 // stripping newline
607 if (line[strlen(line) -1] == '\n') line[strlen(line) -1] = '\0';
608 char *out = malloc(strlen(line) + 1);
609 strcpy(out, line);
610
611 return out;
612}
