118double gkn_mm_score(const gkn_mm mm, const char *seq, int pos, int end) {
119 double p = 0;
120 if (pos < mm->k) pos = mm->k;
121 for (int i = pos; i < end - mm->k +2; i++) {
122 int idx = gkn_ntindex(seq, i, mm->k);
123 if (idx != -1) p += mm->score[idx];
124 }
125 return p;

128double * gkn_mm_cache(const gkn_mm mm, const char *seq) {
129 int len = strlen(seq);
130 double *score = malloc(sizeof(double) * len);
131 for (int i = 0; i < mm->k; i++) score[i] = 0;
132 for (int i = mm->k; i < len; i++) {
133 int idx = gkn_ntindex(seq, i, mm->k);
134 if (idx == -1) score[i] = score[i-1];
135 else score[i] = score[i-1] + mm->score[idx];
136 }
137 return score;
138}

146static double find_tail(double val, int x) {
147 double lo = 0;
148 double hi = 1000;
149 double m;
150
151 while (hi - lo > 1) {
152 m = (hi + lo) / 2;
153 double p = 1 / m;
154 double f = pow(1-p, x-1) * p;
155 if (f < val) lo += (m - lo) / 2;
156 else hi -= (hi - m) / 2;
157 }
158
159 return m;
160}

168gkn_len gkn_len_read(gkn_pipe io) {
169 char *line = NULL;
170 double *score = NULL;
171 double p;
172 int idx = 0;
173 int size;
174 char name[64];
175
176 // read probabilities
177 while ((line = gkn_readline(io)) != NULL) {
178 if (line[0] == '%') {
179 assert(sscanf(line, "%% LEN %s %d", name, &size) == 2);
180 score = malloc(sizeof(double) * size);
181 free(line);
182 } else if (sscanf(line, "%lf", &p) == 1) {
183 score[idx] = p;
184 idx++;
185 free(line);
186 } else {
187 free(line);
188 }
189 }
190
191 gkn_len model = malloc(sizeof(struct gkn_LEN));
192 model->name = malloc(strlen(name)+1);
193 strcpy(model->name, name);
194 model->score = score;
195 model->size = size;
196 model->tail = find_tail(score[size-1], size);
197
198 // convert probabilities to scores
199 double expect = (double) 1 / model->size;
200 for (int i = 0; i < size; i++) {
201 score[i] = log(score[i]/expect) / log(2); // divide by zero?
202 }
203
204 return model;
205}

207double gkn_len_score(const gkn_len len, int x) {
208 assert(x >= 0);
209 if (x >= len->size) {
210 double p = 1 / len->tail;
211 double q = pow(1-p, x-1) * p;
212 double expect = (double)1 / len->size;
213 double s = log(q/expect) / log(2);
214 return s;
215 } else {
216 return len->score[x];
217 }
218}
