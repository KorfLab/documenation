13int gkn_ntindex(const char *seq, int off, int k) {
14 int idx = 0;
15 for (int i = 0; i < k; i++) {
16 switch (seq[off+i]) {
17 case 'A': case 'a': idx += pow(4, (k -i -1)) * 0; break;
18 case 'C': case 'c': idx += pow(4, (k -i -1)) * 1; break;
19 case 'G': case 'g': idx += pow(4, (k -i -1)) * 2; break;
20 case 'T': case 't': idx += pow(4, (k -i -1)) * 3; break;
21 default: return -1;
22 }
23 }
24 return idx;
25}

89gkn_fasta gkn_fasta_read(gkn_pipe io) {
90
91 // check for fasta header
92 char c = fgetc(io->stream);
93 ungetc(c, io->stream);
94 if (c == EOF || (unsigned char)c == 255) return NULL;
95 if (c != '>') gkn_exit("fasta? %c %d", c, (int)c);
96
97 // def
98 char *def = gkn_readline(io);
99
100 // seq
101 gkn_vec lines = gkn_vec_new();
102 while (1) {
103 char c = fgetc(io->stream);
104 ungetc(c, io->stream);
105 if (c == EOF || (unsigned char)c == 255 || c == '>' ) break;
106 char *line = gkn_readline(io);
107 if (line == NULL) break;
108 gkn_vec_push(lines, line);
109 }
110
111 int letters = 0;
112 for (int i = 0; i < lines->size; i++) {
113 char *line = lines->elem[i];
114 letters += strlen(line);
115 }
116
117 char *seq = malloc(letters + 1);
118 int off = 0;
119 for (int i = 0; i < lines->size; i++) {
120 char *line = lines->elem[i];
121 strcpy(seq + off, line);
122 off += strlen(line);
123 }
124
125 // clean up
126 for (int i = 0; i < lines->size; i++) {
127 free(lines->elem[i]);
128 }
129 gkn_vec_free(lines);
130
131 // return object
132 gkn_fasta ff = malloc(sizeof(struct gkn_FASTA));
133 ff->def = def;
134 ff->seq = seq;
135 ff->length = strlen(seq);
